[requires]
boost/1.75.0
openssl/1.1.1i
cpprestsdk/2.10.16
websocketpp/0.8.2
zlib/1.2.11

[imports]
bin, *.dll -> ./bin
lib, *.so* -> ./lib
lib, *.dylib* -> ./lib

[options]
boost:shared=False
# see FIND_BOOST_COMPONENTS in cmake/NmosCppCommon.cmake for the short list of Boost components that are required
# however since it's hard to specify a platform/compiler-specific list to install/build with the Conan recipe
# instead exclude those that cause problems, e.g. since https://github.com/conan-io/conan-center-index/pull/3999
# first up, Boost.Fiber on macOS; see https://stackoverflow.com/a/62445652
boost:without_fiber=True
# Boost.Nowide, added in Boost 1.73.0, also isn't built on macOS
# Boost.JSON, added in Boost 1.75.0, also isn't built on macOS
# see https://github.com/conan-io/conan-center-index/issues/4097
boost:without_nowide=True
boost:without_json=True
